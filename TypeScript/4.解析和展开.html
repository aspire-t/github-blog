# 解构和展开

# 解构

解构是变量申明的一种语法糖 分为数字解构和对象解构

# 数字解构

let input: [number, number] = [1, 2]

function fun([first, second]: [number, number]) {
  console.log(first)
  console.log(second)
}

fun(input)

let [first, ...rest] = [1, 2, 3, 4]
console.log(first)
console.log(rest)

let [first, ...rest] = [1, 2, 3, 4]
console.log(first)
console.log(rest)

let [, second, , fourth] = [1, 2, 3, 4]
console.log(second)
console.log(fourth)

let [, second, , fourth] = [1, 2, 3, 4]
console.log(second)
console.log(fourth)

# 对象解构

let o = {
  a: 'foo',
  b: 12,
  c: 'bar'
}

let { a, b } = o

let { a, ...rest } = o

// 给取的变量取别名可以用下面的方法：
let { a: newName1, b: newName2 } = o
// 上面的写法表达的就是下面写法的意思
let newName1 = o.a
let newName2 = o.b
// 但是这种写法ts会造成混乱

let o = {
  a: 'foo',
  b: 12,
  c: 'bar'
}
let {a,b}:{a:string,b:number} = o;

函数声明

// 函数默认值
function fun2(par: { a: string, b?: number }) {
  let { a, b = 1001 } = par
}

# 展开

数组的展开

let first = [1, 2]
let second = [3, 4]
let third = [0, ...first, ...second, 5]// 这是一个浅拷贝
console.log(third) // [ 0, 1, 2, 3, 4, 5 ]

对象的展开

let defaults = {
  food: 'spicy',
  price: '$10',
  ambiance: 'noisy'
}

let search = { ...defaults, food: 'rich' }
console.log(search) // { food: 'rich', price: '$10', ambiance: 'noisy' }

let defaults = {
  food: 'spicy',
  price: '$10',
  ambiance: 'noisy'
}

let search = {  food: 'rich' ,...defaults}
console.log(search) // { food: 'spicy', price: '$10', ambiance: 'noisy' }
